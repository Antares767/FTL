```python
# –ë–ª–æ–∫ 3.2.3: –†–æ–±–æ—Ç—ã –¥–æ –∏ –ø–æ—Å–ª–µ —Ä–∞–±–æ—Ç—ã
# –ü–æ–∫–∞–∑—ã–≤–∞–µ–º, –∫–∞–∫ —Ä–æ–±–æ—Ç—ã –∏–∑–º–µ–Ω—è—é—Ç—Å—è!

import matplotlib.pyplot as plt
import numpy as np

class Robot:
    """–†–æ–±–æ—Ç, –∫–æ—Ç–æ—Ä—ã–π —É–º–µ–µ—Ç —Ä–∏—Å–æ–≤–∞—Ç—å —Å–µ–±—è –Ω–∞ —ç–∫—Ä–∞–Ω–µ"""
    
    def __init__(self, name, start_x=0, start_y=0, color="blue"):
        self.name = name
        self.x = start_x
        self.y = start_y
        self.start_x = start_x
        self.start_y = start_y
        self.color = color
        self.steps_made = 0
        self.path = [(start_x, start_y)]  # –¥–ª—è –±—É–¥—É—â–µ–≥–æ draw_path()
        print(f"ü§ñ {name} ({color}) —Å–æ–∑–¥–∞–Ω –≤ ({start_x}, {start_y})")
    
    def draw(self, size=100):
        """–†–∏—Å—É–µ—Ç —Ä–æ–±–æ—Ç–∞ –Ω–∞ —Ç–µ–∫—É—â–µ–º –≥—Ä–∞—Ñ–∏–∫–µ"""
        plt.scatter(self.x, self.y, 
                   color=self.color, 
                   s=size, 
                   alpha=0.8,
                   label=self.name)
        
        plt.annotate(f'{self.name}\n({self.x},{self.y})', 
                    (self.x, self.y), 
                    xytext=(10, 10), 
                    textcoords='offset points',
                    fontsize=8,
                    bbox=dict(boxstyle="round,pad=0.3", 
                             facecolor=self.color, 
                             alpha=0.3))
    
    def move_right(self, steps=1):
        """–î–≤–∏–∂–µ–Ω–∏–µ –≤–ø—Ä–∞–≤–æ –Ω–∞ —É–∫–∞–∑–∞–Ω–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —à–∞–≥–æ–≤"""
        if steps <= 0:
            print(f"‚ùå {self.name}: –Ω–µ–ª—å–∑—è –¥–≤–∏–≥–∞—Ç—å—Å—è –Ω–∞ {steps} —à–∞–≥–æ–≤!")
            return
        
        for step in range(steps):
            self.x += 1
            self.steps_made += 1
            self.path.append((self.x, self.y))
            print(f"    ‚û°Ô∏è {self.name} —à–∞–≥ {step+1} ‚Üí ({self.x}, {self.y})")

# –¢–µ—Å—Ç: –†–æ–±–æ—Ç—ã –¥–æ –∏ –ø–æ—Å–ª–µ —Ä–∞–±–æ—Ç—ã
print("üé® –¢–ï–°–¢: –†–æ–±–æ—Ç—ã –¥–æ –∏ –ø–æ—Å–ª–µ —Ä–∞–±–æ—Ç—ã")

# –°–æ–∑–¥–∞–µ–º —Ä–æ–±–æ—Ç–æ–≤ –¥–ª—è —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–∞
work_robots = [
    Robot("–†–∞–±–æ—á–∏–π-1", 2, 2, "purple"),
    Robot("–†–∞–±–æ—á–∏–π-2", 8, 3, "cyan"),
    Robot("–†–∞–±–æ—á–∏–π-3", 5, 7, "magenta")
]

# –ì—Ä–∞—Ñ–∏–∫ –î–û —Ä–∞–±–æ—Ç—ã
plt.figure(figsize=(15, 5))

# –ü–æ–¥–≥—Ä–∞—Ñ–∏–∫ 1: –î–û
plt.subplot(1, 2, 1)
for robot in work_robots:
    robot.draw()

plt.grid(True, alpha=0.3)
plt.xlim(0, 12)
plt.ylim(0, 12)
plt.title("–î–û —Ä–∞–±–æ—Ç—ã")
plt.xlabel("X")
plt.ylabel("Y")
plt.legend()

# –ó–∞—Å—Ç–∞–≤–ª—è–µ–º —Ä–æ–±–æ—Ç–æ–≤ —Ä–∞–±–æ—Ç–∞—Ç—å
steps = [2, 3, 4]
for i, robot in enumerate(work_robots):
    robot.move_right(steps[i])
    print(f"‚úÖ {robot.name} –ø—Ä–æ—à–µ–ª {steps[i]} —à–∞–≥–æ–≤ –≤–ø—Ä–∞–≤–æ")

# –ü–æ–¥–≥—Ä–∞—Ñ–∏–∫ 2: –ü–û–°–õ–ï
plt.subplot(1, 2, 2)

# TODO: –ù–∞—Ä–∏—Å—É–π—Ç–µ —Ä–æ–±–æ—Ç–æ–≤ –ø–æ—Å–ª–µ —Ä–∞–±–æ—Ç—ã

plt.xlabel("X")

plt.legend()

plt.tight_layout()
plt.show()

print("\nüéâ –ú–æ–¥—É–ª—å –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏ –æ—Å–≤–æ–µ–Ω! –†–æ–±–æ—Ç—ã —Ç–µ–ø–µ—Ä—å –≤–∏–¥–Ω—ã!")
```
